# Purpose

The purpose of this change is to rename the Side Navigation item prop `subnav` to `subNav` for consistency with the camel-cased styled component.

## Side Navigation

Update `subnav` item prop to `subNav`

```diff
const nav = [
  {
    title: 'Colors',
    itemId: '#level1.1',
-   subnav: [
+   subNav: [
      {
        title: 'Primary',
        itemId: '#level1.1.1',
      },
      {
        title: 'Shades',
        itemId: '#level1.1.2',
-       subnav: [
+       subNav: [
          {
            title: 'Dark',
            itemId: '#level1.1.2.1',
          },
          {
            title: 'Light',
            itemId: '#level1.1.2.2',
          },
        ],
      },
    ],
  },
  {
    title: 'Sizing',
    itemId: '#level1.2',
  },
  {
    title: 'Typography',
    itemId: '#level1.3',
  },
];

export default () => {
  const [location, setLocation] = useState('#level1.1.1');
  return (
    <Navigation
      items={nav}
      activeItemId={location}
      onChange={({item}) => setLocation(item.itemId)}
    />
  );
};
```
